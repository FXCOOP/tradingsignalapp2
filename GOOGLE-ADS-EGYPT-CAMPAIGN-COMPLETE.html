Step 5: Postback Matching Logic

// File: src/app/api/postback/exness/route.ts export async function POST(request: NextRequest) { const postbackData = await request.json(); const { partner_id, event_type, ftd_amount, user_id: exness_user_id } = postbackData; // ENHANCED: Find user by GCLID (most accurate) // Exness doesn't send GCLID, but we can match by partner_id + timing const { data: clickData } = await supabaseAdmin .from('exness_clicks') .select('user_id, gclid, utm_campaign') .eq('partner_id', partner_id) .order('clicked_at', { ascending: false }) .limit(1) .single(); if (clickData) { // Record conversion with GCLID for Google Ads attribution await supabaseAdmin.from('exness_conversions').insert({ user_id: clickData.user_id, partner_id, event_type, ftd_amount, gclid: clickData.gclid, // ← Link back to Google Ads utm_campaign: clickData.utm_campaign, exness_user_id, raw_postback_data: postbackData, processed: false }); // Auto-upgrade user await supabaseAdmin .from('users') .update({ has_broker_account: true, broker_name: 'Exness', broker_verified_at: new Date().toISOString() }) .eq('id', clickData.user_id); // OPTIONAL: Send conversion back to Google Ads // (requires Google Ads API setup) if (clickData.gclid) { await sendGoogleAdsConversion({ gclid: clickData.gclid, conversion_name: 'Exness_Deposit_Egypt', conversion_value: ftd_amount, currency: 'USD' }); } } }
