// src/app/api/daily-content/route.ts import { NextRequest, NextResponse } from 'next/server' import OpenAI from 'openai' import { prisma } from '@/lib/prisma' const openai = new OpenAI({ apiKey: process.env.OPENAI_API_KEY, }) export async function POST(request: NextRequest) { try { // Generate signals using OpenAI const signalsPrompt = `Generate 6 trading signals for GCC markets today. Markets: TASI (Saudi), ADX (UAE), DFM (Dubai), QE (Qatar), Forex (USD/AED, EUR/AED, GBP/AED) For each signal provide: - symbol: string (e.g., "TASI:2222" for Saudi Aramco) - type: "BUY" | "SELL" | "STRONG_BUY" | "STRONG_SELL" - entryPrice: number - targetPrice: number - stopLoss: number - confidence: number (75-95) - timeframe: "1H" | "4H" | "1D" | "1W" - reasoning: string (2-3 sentences) - riskReward: number (calculate from entry/target/SL) Return ONLY valid JSON array, no markdown.` const completion = await openai.chat.completions.create({ model: process.env.OPENAI_MODEL || "gpt-4o", messages: [{ role: "user", content: signalsPrompt }], temperature: 0.7, max_tokens: 3000, }) const signals = JSON.parse(completion.choices[0].message.content || '[]') // Save to database const savedSignals = await Promise.all( signals.map((signal: any) => prisma.signal.create({ data: { symbol: signal.symbol, type: signal.type, entryPrice: signal.entryPrice, targetPrice: signal.targetPrice, stopLoss: signal.stopLoss, confidence: signal.confidence, timeframe: signal.timeframe, reasoning: signal.reasoning, riskReward: signal.riskReward, }, }) ) ) return NextResponse.json({ success: true, signals: savedSignals, generated: new Date().toISOString(), }) } catch (error: any) { console.error('Signal generation error:', error) return NextResponse.json( { success: false, error: error.message }, { status: 500 } ) } }
