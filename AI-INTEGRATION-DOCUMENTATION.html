// Backend API Endpoint import OpenAI from 'openai'; const openai = new OpenAI({ apiKey: process.env.OPENAI_API_KEY, }); // Generate trading signals async function generateTradingSignals() { const prompt = ` Generate 6-8 high-quality trading signals for GCC markets today. Include: - GCC stocks (Saudi SABIC, UAE Emirates NBD, Qatar QNB) - Forex pairs (AED/USD, SAR/USD, QAR/USD, EUR/USD) - Commodities (Gold, Silver, Crude Oil) - Crypto (Bitcoin, Ethereum) For each signal provide: - Symbol name, Signal type, Entry price - Target price, Stop loss, Confidence level - Timeframe, Technical reasoning, Risk/reward `; const response = await openai.chat.completions.create({ model: "gpt-4o-mini", // 60x cheaper than GPT-4! messages: [{ role: "user", content: prompt }], temperature: 0.7, max_tokens: 2500, }); return JSON.parse(response.choices[0].message.content); }
